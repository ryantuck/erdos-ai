      (∀ A : Set ℕ,
      (∃ c : ℝ, c > 0 ∧ ∃ N₀ : ℕ, ∀ N : ℕ, N ≥ N₀ →
        (countingFunction A N : ℝ) ≥ c * Real.log (N : ℝ)) →
