    12→/-!
    13→# Hausdorff distance
    14→
    15→The Hausdorff distance on subsets of a metric (or emetric) space.
    16→
    17→Given two subsets `s` and `t` of a metric space, their Hausdorff distance is the smallest `d`
    18→such that any point of `s` is within `d` of a point in `t`, and conversely. This quantity
    19→is often infinite (think of `s` bounded and `t` unbounded), and therefore better
    20→expressed in the setting of emetric spaces.
    21→
    22→## Main definitions
    23→
    24→This file introduces:
    25→* `Metric.infEDist x s`, the infimum edistance of a point `x` to a set `s` in an emetric space
    26→* `Metric.hausdorffEDist s t`, the Hausdorff edistance of two sets in an emetric space
    27→* Versions of these notions on metric spaces, called respectively `Metric.infDist`
    28→  and `Metric.hausdorffDist`
    29→
    30→## Main results
    31→* `infEDist_closure`: the edistance to a set and its closure coincide
    32→* `Metric.mem_closure_iff_infEDist_zero`: a point `x` belongs to the closure of `s` iff
    33→  `infEDist x s = 0`
    34→* `IsCompact.exists_infEDist_eq_edist`: if `s` is compact and non-empty, there exists a point `y`
    35→  which attains this edistance
    36→* `IsOpen.exists_iUnion_isClosed`: every open set `U` can be written as the increasing union
    37→  of countably many closed subsets of `U`
    38→
    39→* `hausdorffEDist_closure`: replacing a set by its closure does not change the Hausdorff edistance
    40→* `hausdorffEDist_zero_iff_closure_eq_closure`: two sets have Hausdorff edistance zero
    41→  iff their closures coincide
    42→* the Hausdorff edistance is symmetric and satisfies the triangle inequality
    43→* in particular, closed sets in an emetric space are an emetric space
    44→  (this is shown in `EMetricSpace.Closeds.emetricSpace`)
    45→
    46→* versions of these notions on metric spaces
    47→* `hausdorffEDist_ne_top_of_nonempty_of_bounded`: if two sets in a metric space
    48→  are nonempty and bounded in a metric space, they are at finite Hausdorff edistance.
    49→
    50→## Tags
    51→metric space, Hausdorff distance
    52→-/

    71→/-! ### Distance of a point to a set as a function into `ℝ≥0∞`. -/
    72→
    73→/-- The minimal edistance of a point to a set -/
    74→def infEDist (x : α) (s : Set α) : ℝ≥0∞ :=
    75→  ⨅ y ∈ s, edist x y
    76→
    77→@[simp]
    78→theorem infEDist_empty : infEDist x ∅ = ∞ :=
    79→  iInf_emptyset
    80→
    81→theorem le_infEDist {d} : d ≤ infEDist x s ↔ ∀ y ∈ s, d ≤ edist x y := by
    82→  simp only [infEDist, le_iInf_iff]
    83→
    84→/-- The edist to a union is the minimum of the edists -/
    85→@[simp]
    86→theorem infEDist_union : infEDist x (s ∪ t) = infEDist x s ⊓ infEDist x t :=
    87→  iInf_union
    88→
    89→@[simp]
    90→theorem infEDist_iUnion (f : ι → Set α) (x : α) : infEDist x (⋃ i, f i) = ⨅ i, infEDist x (f i) :=
    91→  iInf_iUnion f _
    92→
    93→lemma infEDist_biUnion {ι : Type*} (f : ι → Set α) (I : Set ι) (x : α) :
    94→    infEDist x (⋃ i ∈ I, f i) = ⨅ i ∈ I, infEDist x (f i) := by simp only [infEDist_iUnion]
    95→
    96→/-- The edist to a singleton is the edistance to the single point of this singleton -/
    97→@[simp]
    98→theorem infEDist_singleton : infEDist x {y} = edist x y :=
    99→  iInf_singleton
   100→
   101→/-- The edist to a set is bounded above by the edist to any of its points -/
   102→theorem infEDist_le_edist_of_mem (h : y ∈ s) : infEDist x s ≤ edist x y :=
   103→  iInf₂_le y h
   104→
   105→/-- If a point `x` belongs to `s`, then its edist to `s` vanishes -/
   106→theorem infEDist_zero_of_mem (h : x ∈ s) : infEDist x s = 0 :=
   107→  nonpos_iff_eq_zero.1 <| @edist_self _ _ x ▸ infEDist_le_edist_of_mem h
   108→
   109→/-- The edist is antitone with respect to inclusion. -/
   110→@[gcongr]
   111→theorem infEDist_anti (h : s ⊆ t) : infEDist x t ≤ infEDist x s :=
   112→  iInf_le_iInf_of_subset h
   113→
   114→/-- The edist to a set is `< r` iff there exists a point in the set at edistance `< r` -/
   115→theorem infEDist_lt_iff {r : ℝ≥0∞} : infEDist x s < r ↔ ∃ y ∈ s, edist x y < r := by
   116→  simp_rw [infEDist, iInf_lt_iff, exists_prop]
   117→
   118→/-- The edist of `x` to `s` is bounded by the sum of the edist of `y` to `s` and
   119→the edist from `x` to `y` -/
   120→theorem infEDist_le_infEDist_add_edist : infEDist x s ≤ infEDist y s + edist x y :=
   121→  calc
   122→    ⨅ z ∈ s, edist x z ≤ ⨅ z ∈ s, edist y z + edist x y :=
   123→      iInf₂_mono fun _ _ => (edist_triangle _ _ _).trans_eq (add_comm _ _)
   124→    _ = (⨅ z ∈ s, edist y z) + edist x y := by simp only [ENNReal.iInf_add]
   125→
   126→theorem infEDist_le_edist_add_infEDist : infEDist x s ≤ edist x y + infEDist y s := by
   127→  rw [add_comm]
   128→  exact infEDist_le_infEDist_add_edist
   129→
   130→theorem edist_le_infEDist_add_ediam (hy : y ∈ s) : edist x y ≤ infEDist x s + Metric.ediam s := by
   131→  simp_rw [infEDist, ENNReal.iInf_add]
   132→  refine le_iInf₂ fun i hi => ?_
   133→  calc
   134→    edist x y ≤ edist x i + edist i y := edist_triangle _ _ _
   135→    _ ≤ edist x i + Metric.ediam s := add_le_add le_rfl (Metric.edist_le_ediam_of_mem hi hy)
   136→
   137→/-- The edist to a set depends continuously on the point -/
   138→@[continuity, fun_prop]
   139→theorem continuous_infEDist : Continuous fun x => infEDist x s :=
   140→  continuous_of_le_add_edist 1 (by simp) <| by
   141→    simp only [one_mul, infEDist_le_infEDist_add_edist, forall₂_true_iff]
   142→
   143→/-- The edist to a set and to its closure coincide -/
   144→theorem infEDist_closure : infEDist x (closure s) = infEDist x s := by
   145→  refine le_antisymm (infEDist_anti subset_closure) ?_
   146→  refine ENNReal.le_of_forall_pos_le_add fun ε εpos h => ?_
   147→  have ε0 : 0 < (ε / 2 : ℝ≥0∞) := by simpa [pos_iff_ne_zero] using εpos
   148→  have : infEDist x (closure s) < infEDist x (closure s) + ε / 2 :=
   149→    ENNReal.lt_add_right h.ne ε0.ne'
   150→  obtain ⟨y : α, ycs : y ∈ closure s, hy : edist x y < infEDist x (closure s) + ↑ε / 2⟩ :=
   151→    infEDist_lt_iff.mp this
   152→  obtain ⟨z : α, zs : z ∈ s, dyz : edist y z < ↑ε / 2⟩ := EMetric.mem_closure_iff.1 ycs (ε / 2) ε0
   153→  calc
   154→    infEDist x s ≤ edist x z := infEDist_le_edist_of_mem zs
   155→    _ ≤ edist x y + edist y z := edist_triangle _ _ _
   156→    _ ≤ infEDist x (closure s) + ε / 2 + ε / 2 := add_le_add (le_of_lt hy) (le_of_lt dyz)
   157→    _ = infEDist x (closure s) + ↑ε := by rw [add_assoc, ENNReal.add_halves]
   158→
   159→/-- A point belongs to the closure of `s` iff its infimum edistance to this set vanishes -/
   160→theorem mem_closure_iff_infEDist_zero : x ∈ closure s ↔ infEDist x s = 0 :=
   161→  ⟨fun h => by
   162→    rw [← infEDist_closure]
   163→    exact infEDist_zero_of_mem h,
   164→   fun h =>
   165→    EMetric.mem_closure_iff.2 fun ε εpos => infEDist_lt_iff.mp <| by rwa [h]⟩
   166→
   167→/-- Given a closed set `s`, a point belongs to `s` iff its infimum edistance to this set vanishes -/
   168→theorem mem_iff_infEDist_zero_of_closed (h : IsClosed s) : x ∈ s ↔ infEDist x s = 0 := by
   169→  rw [← mem_closure_iff_infEDist_zero, h.closure_eq]
   170→
   171→/-- The infimum edistance of a point to a set is positive if and only if the point is not in the
   172→closure of the set. -/
   173→theorem infEDist_pos_iff_notMem_closure {x : α} {E : Set α} :
   174→    0 < infEDist x E ↔ x ∉ closure E := by
   175→  rw [mem_closure_iff_infEDist_zero, pos_iff_ne_zero]
   176→
   177→theorem infEDist_closure_pos_iff_notMem_closure {x : α} {E : Set α} :
   178→    0 < infEDist x (closure E) ↔ x ∉ closure E := by
   179→  rw [infEDist_closure, infEDist_pos_iff_notMem_closure]
   180→
   181→theorem exists_real_pos_lt_infEDist_of_notMem_closure {x : α} {E : Set α} (h : x ∉ closure E) :
   182→    ∃ ε : ℝ, 0 < ε ∧ ENNReal.ofReal ε < infEDist x E := by
   183→  rw [← infEDist_pos_iff_notMem_closure, ENNReal.lt_iff_exists_real_btwn] at h
   184→  rcases h with ⟨ε, ⟨_, ⟨ε_pos, ε_lt⟩⟩⟩
   185→  exact ⟨ε, ⟨ENNReal.ofReal_pos.mp ε_pos, ε_lt⟩⟩
   186→
   187→theorem disjoint_closedEBall_of_lt_infEDist {r : ℝ≥0∞} (h : r < infEDist x s) :
   188→    Disjoint (EMetric.closedBall x r) s := by
   189→  rw [disjoint_left]
   190→  intro y hy h'y
   191→  apply lt_irrefl (infEDist x s)
   192→  calc
   193→    infEDist x s ≤ edist x y := infEDist_le_edist_of_mem h'y
   194→    _ ≤ r := by rwa [EMetric.mem_closedBall, edist_comm] at hy
   195→    _ < infEDist x s := h

   197→/-- The infimum edistance is invariant under isometries -/
   198→theorem infEDist_image (hΦ : Isometry Φ) : infEDist (Φ x) (Φ '' t) = infEDist x t := by
   199→  simp only [infEDist, iInf_image, hΦ.edist_eq]

   201→@[to_additive (attr := simp)]
   202→theorem infEDist_smul {M} [SMul M α] [IsIsometricSMul M α] (c : M) (x : α) (s : Set α) :
   203→    infEDist (c • x) (c • s) = infEDist x s :=

   206→theorem _root_.IsOpen.exists_iUnion_isClosed {U : Set α} (hU : IsOpen U) :
   207→    ∃ F : ℕ → Set α, (∀ n, IsClosed (F n)) ∧ (∀ n, F n ⊆ U) ∧ ⋃ n, F n = U ∧ Monotone F := by
   208→  obtain ⟨a, a_pos, a_lt_one⟩ : ∃ a : ℝ≥0∞, 0 < a ∧ a < 1 := exists_between zero_lt_one
   209→  let F := fun n : ℕ => (fun x => infEDist x Uᶜ) ⁻¹' Ici (a ^ n)
   210→  have F_subset : ∀ n, F n ⊆ U := fun n x hx ↦ by
   211→    by_contra h
   212→    have : infEDist x Uᶜ ≠ 0 := ((ENNReal.pow_pos a_pos _).trans_le hx).ne'
   213→    exact this (infEDist_zero_of_mem h)
   214→  refine ⟨F, fun n => IsClosed.preimage continuous_infEDist isClosed_Ici, F_subset, ?_, ?_⟩
   215→  · show ⋃ n, F n = U
   216→    refine Subset.antisymm (by simp only [iUnion_subset_iff, F_subset, forall_const]) fun x hx => ?_
   217→    have : x ∉ Uᶜ := by simpa using hx
   218→    rw [mem_iff_infEDist_zero_of_closed hU.isClosed_compl] at this
   219→    have B : 0 < infEDist x Uᶜ := by simpa [pos_iff_ne_zero] using this
   220→    have : Filter.Tendsto (fun n => a ^ n) atTop (𝓝 0) :=
   221→      ENNReal.tendsto_pow_atTop_nhds_zero_of_lt_one a_lt_one
   222→    rcases ((tendsto_order.1 this).2 _ B).exists with ⟨n, hn⟩
   223→    simp only [mem_iUnion]
   224→    exact ⟨n, hn.le⟩
   225→  show Monotone F
   226→  intro m n hmn x hx
   227→  simp only [F, mem_Ici, mem_preimage] at hx ⊢
   228→  apply le_trans (pow_le_pow_right_of_le_one' a_lt_one.le hmn) hx

   230→theorem _root_.IsCompact.exists_infEDist_eq_edist (hs : IsCompact s) (hne : s.Nonempty) (x : α) :
   231→    ∃ y ∈ s, infEDist x s = edist x y := by
   232→  have A : Continuous fun y => edist x y := continuous_const.edist continuous_id
   233→  obtain ⟨y, ys, hy⟩ := hs.exists_isMinOn hne A.continuousOn
   234→  exact ⟨y, ys, le_antisymm (infEDist_le_edist_of_mem ys) (by rwa [le_infEDist])⟩
   235→
   236→theorem exists_pos_forall_lt_edist (hs : IsCompact s) (ht : IsClosed t) (hst : Disjoint s t) :
   237→    ∃ r : ℝ≥0, 0 < r ∧ ∀ x ∈ s, ∀ y ∈ t, (r : ℝ≥0∞) < edist x y := by
   238→  rcases s.eq_empty_or_nonempty with (rfl | hne)

   244→  rcases ENNReal.lt_iff_exists_nnreal_btwn.1 this with ⟨r, h₀, hr⟩
   245→  exact ⟨r, ENNReal.coe_pos.mp h₀, fun y hy z hz => hr.trans_le <| le_infEDist.1 (h hy) z hz⟩

   254→/-! ### The Hausdorff distance as a function into `ℝ≥0∞`. -/
   255→
   256→/-- The Hausdorff edistance between two sets is the smallest `r` such that each set
   257→is contained in the `r`-neighborhood of the other one -/
   258→irreducible_def hausdorffEDist {α : Type u} [PseudoEMetricSpace α] (s t : Set α) : ℝ≥0∞ :=
   259→  (⨆ x ∈ s, infEDist x t) ⊔ ⨆ y ∈ t, infEDist y s
   260→
   261→section HausdorffEDist
   262→
   263→variable [PseudoEMetricSpace α] [PseudoEMetricSpace β] {x y : α} {s t u : Set α} {Φ : α → β}
   264→
   265→/-- The Hausdorff edistance of a set to itself vanishes. -/
   266→@[simp]
   267→theorem hausdorffEDist_self : hausdorffEDist s s = 0 := by
   268→  simp only [hausdorffEDist_def, sup_idem, ENNReal.iSup_eq_zero]
   269→  exact fun x hx => infEDist_zero_of_mem hx
   270→
   271→/-- The Hausdorff edistances of `s` to `t` and of `t` to `s` coincide. -/
   272→theorem hausdorffEDist_comm : hausdorffEDist s t = hausdorffEDist t s := by
   273→  simp only [hausdorffEDist_def]; apply sup_comm
   274→
   275→/-- Bounding the Hausdorff edistance by bounding the edistance of any point
   276→in each set to the other set -/
   277→theorem hausdorffEDist_le_of_infEDist {r : ℝ≥0∞} (H1 : ∀ x ∈ s, infEDist x t ≤ r)
   278→    (H2 : ∀ x ∈ t, infEDist x s ≤ r) : hausdorffEDist s t ≤ r := by
   279→  simp only [hausdorffEDist_def, sup_le_iff, iSup_le_iff]
   280→  exact ⟨H1, H2⟩
   281→
   282→/-- Bounding the Hausdorff edistance by exhibiting, for any point in each set,
   283→another point in the other set at controlled distance -/
   284→theorem hausdorffEDist_le_of_mem_edist {r : ℝ≥0∞} (H1 : ∀ x ∈ s, ∃ y ∈ t, edist x y ≤ r)
   285→    (H2 : ∀ x ∈ t, ∃ y ∈ s, edist x y ≤ r) : hausdorffEDist s t ≤ r := by
   286→  refine hausdorffEDist_le_of_infEDist (fun x xs ↦ ?_) (fun x xt ↦ ?_)
   287→  · rcases H1 x xs with ⟨y, yt, hy⟩
   288→    exact le_trans (infEDist_le_edist_of_mem yt) hy
   289→  · rcases H2 x xt with ⟨y, ys, hy⟩
   290→    exact le_trans (infEDist_le_edist_of_mem ys) hy
   291→
   292→/-- The distance to a set is controlled by the Hausdorff distance. -/
   293→theorem infEDist_le_hausdorffEDist_of_mem (h : x ∈ s) : infEDist x t ≤ hausdorffEDist s t := by
   294→  rw [hausdorffEDist_def]
   295→  refine le_trans ?_ le_sup_left
   296→  exact le_iSup₂ (α := ℝ≥0∞) x h
   297→
   298→/-- If the Hausdorff distance is `< r`, then any point in one of the sets has
   299→a corresponding point at distance `< r` in the other set. -/
   300→theorem exists_edist_lt_of_hausdorffEDist_lt {r : ℝ≥0∞} (h : x ∈ s) (H : hausdorffEDist s t < r) :
   301→    ∃ y ∈ t, edist x y < r :=
   302→  infEDist_lt_iff.mp <|
   303→    calc
   304→      infEDist x t ≤ hausdorffEDist s t := infEDist_le_hausdorffEDist_of_mem h
   305→      _ < r := H
   306→
   307→/-- The distance from `x` to `s` or `t` is controlled in terms of the Hausdorff distance
   308→between `s` and `t`. -/
   309→theorem infEDist_le_infEDist_add_hausdorffEDist :
   310→    infEDist x t ≤ infEDist x s + hausdorffEDist s t :=
   311→  ENNReal.le_of_forall_pos_le_add fun ε εpos h => by
   312→    have ε0 : (ε / 2 : ℝ≥0∞) ≠ 0 := by simpa [pos_iff_ne_zero] using εpos
   313→    have : infEDist x s < infEDist x s + ε / 2 :=
   314→      ENNReal.lt_add_right (ENNReal.add_lt_top.1 h).1.ne ε0
   315→    obtain ⟨y : α, ys : y ∈ s, dxy : edist x y < infEDist x s + ↑ε / 2⟩ := infEDist_lt_iff.mp this
   316→    have : hausdorffEDist s t < hausdorffEDist s t + ε / 2 :=
   317→      ENNReal.lt_add_right (ENNReal.add_lt_top.1 h).2.ne ε0
   318→    obtain ⟨z : α, zt : z ∈ t, dyz : edist y z < hausdorffEDist s t + ↑ε / 2⟩ :=
   319→      exists_edist_lt_of_hausdorffEDist_lt ys this
   320→    calc
   321→      infEDist x t ≤ edist x z := infEDist_le_edist_of_mem zt
   322→      _ ≤ edist x y + edist y z := edist_triangle _ _ _
   323→      _ ≤ infEDist x s + ε / 2 + (hausdorffEDist s t + ε / 2) := add_le_add dxy.le dyz.le
   324→      _ = infEDist x s + hausdorffEDist s t + ε := by
   325→        rw [add_add_add_comm, ENNReal.add_halves]
   326→
   327→/-- The Hausdorff edistance is invariant under isometries. -/
   328→theorem hausdorffEDist_image (h : Isometry Φ) :
   329→    hausdorffEDist (Φ '' s) (Φ '' t) = hausdorffEDist s t := by
   330→  simp only [hausdorffEDist_def, iSup_image, infEDist_image h]
   331→
   332→/-- The Hausdorff distance is controlled by the diameter of the union. -/
   333→theorem hausdorffEDist_le_ediam (hs : s.Nonempty) (ht : t.Nonempty) :
   334→    hausdorffEDist s t ≤ Metric.ediam (s ∪ t) := by
   335→  rcases hs with ⟨x, xs⟩
   336→  rcases ht with ⟨y, yt⟩
   337→  refine hausdorffEDist_le_of_mem_edist ?_ ?_
   338→  · intro z hz
   339→    exact ⟨y, yt, Metric.edist_le_ediam_of_mem (subset_union_left hz) (subset_union_right yt)⟩
   340→  · intro z hz
   341→    exact ⟨x, xs, Metric.edist_le_ediam_of_mem (subset_union_right hz) (subset_union_left xs)⟩
   342→
   343→/-- The Hausdorff distance satisfies the triangle inequality. -/
   344→theorem hausdorffEDist_triangle : hausdorffEDist s u ≤ hausdorffEDist s t + hausdorffEDist t u := by
   345→  rw [hausdorffEDist_def]
   346→  simp only [sup_le_iff, iSup_le_iff]
   347→  constructor
   348→  · change ∀ x ∈ s, infEDist x u ≤ hausdorffEDist s t + hausdorffEDist t u
   349→    exact fun x xs =>
   350→      calc
   351→        infEDist x u ≤ infEDist x t + hausdorffEDist t u :=
   352→          infEDist_le_infEDist_add_hausdorffEDist
   353→        _ ≤ hausdorffEDist s t + hausdorffEDist t u := by grw [infEDist_le_hausdorffEDist_of_mem xs]
   354→  · change ∀ x ∈ u, infEDist x s ≤ hausdorffEDist s t + hausdorffEDist t u
   355→    exact fun x xu =>
   356→      calc
   357→        infEDist x s ≤ infEDist x t + hausdorffEDist t s :=
   358→          infEDist_le_infEDist_add_hausdorffEDist
   359→        _ ≤ hausdorffEDist u t + hausdorffEDist t s := by grw [infEDist_le_hausdorffEDist_of_mem xu]
   360→        _ = hausdorffEDist s t + hausdorffEDist t u := by simp [hausdorffEDist_comm, add_comm]
   361→
   362→/-- Two sets are at zero Hausdorff edistance if and only if they have the same closure. -/

   365→  simp only [hausdorffEDist_def, ENNReal.max_eq_zero_iff, ENNReal.iSup_eq_zero, ← subset_def,

   376→  refine le_antisymm ?_ ?_

   379→      _ = hausdorffEDist s t := by simp [hausdorffEDist_comm]

   382→      _ = hausdorffEDist (closure s) t := by simp
