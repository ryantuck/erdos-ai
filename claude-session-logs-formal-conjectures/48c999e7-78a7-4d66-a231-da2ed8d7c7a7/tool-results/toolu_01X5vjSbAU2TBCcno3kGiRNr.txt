    11→/-!
    12→# Subgraphs of a simple graph
    13→
    14→A subgraph of a simple graph consists of subsets of the graph's vertices and edges such that the
    15→endpoints of each edge are present in the vertex subset. The edge subset is formalized as a
    16→sub-relation of the adjacency relation of the simple graph.
    17→
    18→## Main definitions
    19→
    20→* `Subgraph G` is the type of subgraphs of a `G : SimpleGraph V`.
    21→
    22→* `Subgraph.neighborSet`, `Subgraph.incidenceSet`, and `Subgraph.degree` are like their
    23→  `SimpleGraph` counterparts, but they refer to vertices from `G` to avoid subtype coercions.
    24→
    25→* `Subgraph.coe` is the coercion from a `G' : Subgraph G` to a `SimpleGraph G'.verts`.
    26→  (In Lean 3 this could not be a `Coe` instance since the destination type depends on `G'`.)
    27→
    28→* `Subgraph.IsSpanning` for whether a subgraph is a spanning subgraph and
    29→  `Subgraph.IsInduced` for whether a subgraph is an induced subgraph.
    30→
    31→* Instances for `Lattice (Subgraph G)` and `BoundedOrder (Subgraph G)`.
    32→
    33→* `SimpleGraph.toSubgraph`: If a `SimpleGraph` is a subgraph of another, then you can turn it
    34→  into a member of the larger graph's `SimpleGraph.Subgraph` type.
    35→
    36→* Graph homomorphisms from a subgraph to a graph (`Subgraph.map_top`) and between subgraphs
    37→  (`Subgraph.map`).
    38→
    39→## Implementation notes
    40→
    41→* Recall that subgraphs are not determined by their vertex sets, so `SetLike` does not apply to
    42→  this kind of subobject.
    43→
    44→## TODO
    45→
    46→* Images of graph homomorphisms as subgraphs.

    57→/-- A subgraph of a `SimpleGraph` is a subset of vertices along with a restriction of the adjacency
    58→relation that is symmetric and is supported by the vertex subset.  They also form a bounded lattice.
    59→
    60→Thinking of `V → V → Prop` as `Set (V × V)`, a set of darts (i.e., half-edges), then
    61→`Subgraph.adj_sub` is that the darts of a subgraph are a subset of the darts of `G`. -/
    62→@[ext]
    63→structure Subgraph {V : Type u} (G : SimpleGraph V) where
    64→  /-- Vertices of the subgraph -/
    65→  verts : Set V
    66→  /-- Edges of the subgraph -/
    67→  Adj : V → V → Prop
    68→  adj_sub : ∀ {v w : V}, Adj v w → G.Adj v w
    69→  edge_vert : ∀ {v w : V}, Adj v w → v ∈ verts
    70→  symm : Symmetric Adj := by aesop_graph
    71→
    72→initialize_simps_projections SimpleGraph.Subgraph (Adj → adj)

    88→  verts := {v, w}
    89→  Adj a b := s(v, w) = s(a, b)

    91→    rw [← G.mem_edgeSet, ← h]

   205→/-- A subgraph is called an *induced subgraph* if vertices of `G'` are adjacent if

   207→def IsInduced (G' : Subgraph G) : Prop :=
   208→  ∀ ⦃v⦄, v ∈ G'.verts → ∀ ⦃w⦄, w ∈ G'.verts → G.Adj v w → G'.Adj v w

   331→instance : SupSet G.Subgraph where
   332→  sSup s :=
   333→    { verts := ⋃ G' ∈ s, verts G'
   334→      Adj := fun a b => ∃ G' ∈ s, Adj G' a b
   335→      adj_sub := by
   336→        rintro a b ⟨G', -, hab⟩
   337→        exact G'.adj_sub hab
   338→      edge_vert := by
   339→        rintro a b ⟨G', hG', hab⟩
   340→        exact Set.mem_iUnion₂_of_mem hG' (G'.edge_vert hab)
   341→      symm := fun a b h => by simpa [adj_comm] using h }
